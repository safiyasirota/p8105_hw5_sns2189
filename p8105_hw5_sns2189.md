``` r
prop_test_function <- function(city_df) {

    city_df %>%

  city_test <-
    prop.test(
      x = city_summary %>% pull(unsolved),
```

``` r
#tidying data
experiment_df <-
  experiment_df %>% 
  unnest(subj_data) %>% 
  mutate(
    arm = as.factor(
      case_when(
      str_starts(file, "con") ~ "control",
      TRUE ~ "experimental"
      )
    ),
    subject_n = case_when(
      arm == "control" ~ as.numeric((str_sub(file, 5, 6))),
      TRUE ~ as.numeric((str_sub(file, 5, 6))) + 10
    )
  ) %>% 
  select(-file) %>% 
  relocate(subject_n, arm) %>% 
  pivot_longer(
    week_1:week_8,
    names_to = "week",
    values_to = "value",
    names_prefix = "week_"
  )
```
